\begin{itemize}
	\item \tophead{Ellipsis}
		\begin{itemize}
			\item \subhead{VP-ellipsis}
				\begin{examples}
					\item John eats bread and Mary does, too.
					\item Cathy read the book by Chomsky but she heard from Peter than John told him that Mary didn't.
				\end{examples}
% 
				\begin{example}
					\begin{enumerate}
						\item[(a)] Holly Golightly won't eat rutabagas.
						\item[(b)] I don't think Fred will, either.
					\end{enumerate}
				\end{example}

				\begin{example}
					Anyone who can, should show me how to wiggle my ears.
				\end{example}
			
			\item \subhead{Antecedent-Contained Deletion}
				\begin{example}
					I visited every country I had to.
				\end{example}
			
			\item \subhead{Sluicing}
				\begin{example}
					I know you talked to someone, but I can't remember who.
				\end{example}
				\begin{examples}
					\item I believe the claim that he bit someone, but they don't know who (CNPC)
					\item That he'll hire someone is possible, but I won't divulge who (Sentential Subject Constraint)
					\item Irv and someone were dancing together, but I don't know who (CSC)
				\end{examples}
			
			\item \subhead{N$'$-deletion}
				\begin{examples}
					\item Sal read Bertie's book, but he hasn't read Richie's, yet.
					\item Sal read Bertie's book about marriage, but he hasn't read Richie's about phrase markers.
				\end{examples}
			
			\item \subhead{Pseudo-Gapping}
				\begin{examples}
					\item Sal eats Beans and Holly does rice.
					\begin{enumerate}
						\item I want to understand VP-ellipsis
						\item I do, too./ $^*$ I do Sluicing.
					\end{enumerate}
				\end{examples}
		\end{itemize}
\end{itemize}
\end{minipage}

\begin{minipage}[t]{0.5\textwidth}
\begin{itemize}
		\item \tophead{Stripping}\\
		closer to gapping than VP-ellipsis
		\begin{itemize}
			\item VP: intersentional, no focus needed
			\item Gapping: ungrammatical in subordinate clauses,\\
			may not precede its antecedent, does not delete entire phrases
		\end{itemize}
% 
		\begin{examples}
			\item I sleep a lot and Peter, too.
			\item I sleep a lot. Peter, too.
			\item *John studied rocks even though not Jane
			\item Jane loves to study rocks, and geography, too.
		\end{examples}

		\item \tophead{Gapping}
		\begin{examples}
			\item Some ate \focus{natto} and others \focus{rice}.
			\item Some ate \focus{Bertie}'s book about \focus{marriage} and \focus{Richie}'s about \focus{divorce}.
			\item I made \focus{Sam} angry at \focus{Beaner} and \focus{Betsy} at \focus{Perseus}.
			\item Some gave \focus{albums} to their spouses, and others \focus{tapes}.
		\end{examples}

		\item \tophead{Right-Node-Raising}
		\begin{examples}
			\item John loves and Mary hates cheese.
			\item Susan suspects but Mary knows that John likes her.
			\item Each year John has a spring and an autumn depression.
		\end{examples}
% 
		contra gapping: targets right node\\
		contra ellipsis: targets non-constituents, gap c-commands its antecedent
\end{itemize}
